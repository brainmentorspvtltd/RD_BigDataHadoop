1. wget https://archive.apache.org/dist/hive/hive-3.1.2/apache-hive-3.1.2-bin.tar.gz
2. tar -xzf apache-hive-3.1.2-bin.tar.gz
3. sudo gedit .bashrc
4. Set Hive path
  - export HIVE_HOME=/path/apache-hive-3.1.2-bin
  - export PATH=$PATH:/path/apache-hive-3.1.2-bin/bin
5. source .bashrc
6. Check Hive version
   - hive --version
7. Create Hive directory in HDFS
  - hdfs dfs -mkdir /user/hive/warehouse
8. Give permissions to warehouse folder
  - hdfs dfs -chmod g+w /user/hive/warehouse
  - hdfs dfs -chmod g+w /tmp
9. Set Hadoop path in hive-env.sh (we have to create this file)
10. cd conf 
    sudo cp hive-env.sh.template hive-env.sh
11. sudo gedit conf/hive-env.sh
    export HADOOP_HOME=$HADOOP_HOME
    export HADOOP_HEAPSIZE=512
    export HIVE_CONF_DIR=$HIVE_HOME/conf
12. Edit hive-site.xml in folder conf

<property>
<name>javax.jdo.option.ConnectionURL</name>
<value>jdbc:derby:;databaseName=/path/apache-hive-2.1.0-bin/metastore_db;create=true</value>
<description>

13. Initialize derby database
    bin/schematool -initSchema -dbType derby
Use sudo if permission error
14. lauch hiveserver and hive
    cd bin
    sudo ./hiveserver2
    Open another terminal tab
    sudo -u hdusr_ ./hive
